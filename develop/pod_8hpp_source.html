319 void POD<TruthModelType>::fillPodMatrix( const wn_type& elements_set)
320 {
321  if( M_use_solutions )
322  {
323  //M_bdf->setRestart( true );
324  int K = M_bdf->timeValues().size()-1;
325  M_pod_matrix.resize( K,K );
326 
327  auto bdfi = M_bdf->deepCopy();
328  auto bdfj = M_bdf->deepCopy();
329 
330  bdfi->setRestart( true );
331  bdfj->setRestart( true );
332  bdfi->setTimeInitial( M_time_initial );
333  bdfj->setTimeInitial( M_time_initial );
334  bdfi->setRestartAtLastSave(false);
335  bdfj->setRestartAtLastSave(false);
336  for ( bdfi->restart(); !bdfi->isFinished(); bdfi->next() )
337  {
338  int i = bdfi->iteration()-1;
339  bdfi->loadCurrent();
340 
341  //here is a barrier. For now, if this barrier is removed,
342  //during the bdfj->restart() the program will wait and the memory increase.
343  //during the init step in bdf, the metadata are read,
344  //and it is during this step that the memory could increase dangerously.
345  Environment::worldComm().barrier();
346 
347  for ( bdfj->restart(); !bdfj->isFinished() && ( bdfj->iteration() < bdfi->iteration() ); bdfj->next() )
348  {
349  int j = bdfj->iteration()-1;
350  bdfj->loadCurrent();
351 
352  M_pod_matrix( i,j ) = M_model->scalarProductForPod( bdfj->unknown( 0 ), bdfi->unknown( 0 ) );
353  M_pod_matrix( j,i ) = M_pod_matrix( i,j );
354  }
355 
356  M_pod_matrix( i,i ) = M_model->scalarProductForPod( bdfi->unknown( 0 ), bdfi->unknown( 0 ) );
357  }
358  }//fill pod matrix with solutions
359  else
360  {
361  //use elements given by CRB
362  int size = elements_set.size();
363  M_pod_matrix.resize( size , size );
364  CHECK( size > 0 )<<" elements set doesn't conatin any element\n";
365  for(int i=0; i<size; i++)
366  {
367  for(int j=i+1; j<size; j++)
368  {
369  M_pod_matrix( i,j ) = M_model->scalarProductForPod( elements_set[i], elements_set[j] );
370  M_pod_matrix( j,i ) = M_pod_matrix( i,j );
371  }
372  M_pod_matrix( i,i ) = M_model->scalarProductForPod( elements_set[i] , elements_set[i] );
373  }
374  }

408  }
409

412 
413  LOG(INFO)<<" matrix wrote in file named PodMatrix \n";
414  }
415

417  {
418  std::ofstream matrix_file;

424  matrix_file<<"# columns: "<<M_pod_matrix.cols()<<"\n";

426  for ( int i=0; i<M_pod_matrix.rows(); i++ )
427  {
428  for ( int j=0; j<M_pod_matrix.cols(); j++ )
429  {
430  matrix_file<< std::setprecision( 16 ) <<M_pod_matrix( i,j )<<" ";
431  }

433  matrix_file<<"\n";
434  }
435 
436  LOG(INFO)<<" matrix wrote in file named PodMatrix \n";
437  matrix_file.close();
438  }

440 
441  eigen_solver.compute( M_pod_matrix ); // solve M_pod_matrix psi = lambda psi
442 
443  int number_of_eigenvalues = eigen_solver.eigenvalues().size();
444  LOG(INFO)<<"Number of eigenvalues : "<<number_of_eigenvalues<<"\n";
445  //we copy eigenvalues in a std::vector beacause it's easier to manipulate it
446  std::vector<double> eigen_values( number_of_eigenvalues );
447 
448  int too_small_index = 0;
449 
450  for ( int i=0; i<number_of_eigenvalues; i++ )
451  {
452  if ( imag( eigen_solver.eigenvalues()[i] )>1e-12 )
453  {
454  throw std::logic_error( "[POD::pod] ERROR : complex eigenvalues were found" );
455  }
456 
457  eigen_values[i]=real( eigen_solver.eigenvalues()[i] );
458  }
459 
460 
461  //tab will contains index of maximum eigenvalues
462  std::vector<int> max_idx;
463  std::vector<double> copy_eigen_values ( eigen_values );
464  int number_of_good_eigenvectors=0;
465  bool go = true;
466  double min_eigenvalue = option(_name="pod.minimum-eigenvalue").template as<double>();
467  //for(int maxmode=0; maxmode<M_Nm; maxmode++)
468  auto max = std::max_element( copy_eigen_values.begin(), copy_eigen_values.end() );
469  double idx = std::distance( copy_eigen_values.begin(), max );
470  if( *max < min_eigenvalue )
471  go=false;
472  while( go )
473  {
474  number_of_good_eigenvectors++;
475  max_idx.push_back( idx );
476  copy_eigen_values[idx]=0;
477  max = std::max_element( copy_eigen_values.begin(), copy_eigen_values.end() );
478  idx = std::distance( copy_eigen_values.begin(), max );
479  if( *max < min_eigenvalue )
480  go = false;
481  if( number_of_good_eigenvectors == M_Nm )
482  go = false;
483  if( number_of_good_eigenvectors == number_of_eigenvalues )
484  go = false;
485  }
486 
487  CHECK( number_of_good_eigenvectors > 0 )<<"The max eigenvalue "<<*max<<" is under the minimum eigenvalue set by the user "<<min_eigenvalue<<" and we have zero eigenvectors !\n";
488 
489  int position_of_largest_eigenvalue=max_idx[0];
490 
491  if( M_Nm == -1 )
492  {
493  //in this case the user doesn't know a priori
494  //what value give to M_Nm
495  //this case appears for when apply-POD-to-WN=true
496  M_Nm = number_of_good_eigenvectors;
497  }
498  if ( M_Nm > number_of_good_eigenvectors && number_of_good_eigenvectors>0 && is_primal )
499  {

501  //the minimum eigenvalue set, we don't find enough eigenvectors
502  M_Nm = number_of_good_eigenvectors;
503  }
504

506  {
507  element_ptrtype mode ( new element_type( M_model->functionSpace() ) );
508  mode->zero();
509 
510  int index=0;
511 
512  if( M_use_solutions )
513  {
514  M_bdf->setRestart( true );
515 
516  for ( M_bdf->restart(); !M_bdf->isFinished(); M_bdf->next() )
517  {
518  M_bdf->loadCurrent();
519  double psi_k = real( eigen_solver.eigenvectors().col( position_of_largest_eigenvalue )[index] );
520  M_bdf->unknown( 0 ).scale( psi_k );
521  mode->add( 1 , M_bdf->unknown( 0 ) );
522  index++;
523  }
524  }//if use solutions
525  else
526  {
527  double eigenvalue = eigen_values[ position_of_largest_eigenvalue ];
528  for(int j=0; j<size; j++)
529  {
530  double psi_k = real( eigen_solver.eigenvectors().col( position_of_largest_eigenvalue )[index] );
531  mode->add( psi_k , elements_set[j] );
532  index++;
533  }
534  }//if use elements set
535 
536 
537  //check
538  auto eigenvector = eigen_solver.eigenvectors().col( position_of_largest_eigenvalue );
539  double eigenvalue = eigen_values[position_of_largest_eigenvalue];
540  auto Aw = M_pod_matrix*eigenvector;
541  auto lambdaw = eigenvalue*eigenvector;
542  double Awnorm = Aw.norm();
543  double lambdawnorm = lambdaw.norm();
544  double eigenvectornorm = eigenvector.norm();
545  CHECK( math::abs(Awnorm - lambdawnorm) < 1e-12 )<<" A w : "<<Awnorm<<" and lambda w : "<<lambdawnorm<<" so math::abs(A w - lambda w) : "<<math::abs(Awnorm - lambdawnorm)<<" -- eigenvalue : "<<eigenvalue<<"\n";
546 
547  if( (i+1) < max_idx.size() )
548  position_of_largest_eigenvalue=max_idx[i+1];
549  ModeSet.push_back( *mode );
550  }// loop on number of modes
551 
552

572  position_of_largest_eigenvalue=max_idx[i];
573  }
574  }

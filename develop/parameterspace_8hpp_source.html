331 
332  LOG( INFO ) <<"total_proc : "<<total_proc;
333  LOG( INFO ) <<"proc : "<<proc;
334  LOG( INFO ) <<"total_number_of_elements : "<<N;

336  LOG( INFO ) <<"extra_elements : "<<extra_elements;

343  int count_reqs=0;
344 
345  int tag=0;

347 
348

350  {
351  std::ifstream file ( file_name );
352  CHECK( file ) << "The file "<<file_name<<" was not found so we can't distribute the sampling on all processors\n";
353  this->readFromFile( file_name );

356  for(int i=0; i<N; i++)
357  {

361 
362 
363  for(int proc_recv_sampling=0; proc_recv_sampling<total_proc; proc_recv_sampling++)
364  {
365  if( proc_recv_sampling < extra_elements )
366  {
367  total_number_of_elements_per_proc=number_of_elements_per_proc+1;
368  shift=proc_recv_sampling*total_number_of_elements_per_proc;
369  }
370  else
371  {

373  total_number_of_elements_per_proc=e;
374  shift=extra_elements*(e+1) + (proc_recv_sampling-extra_elements)*e;
375  }
376 
377  if( proc_recv_sampling != master_proc )
378  {

380  {
381  int idx=local_element+shift;
382  super::push_back( temporary_sampling[idx] );
383  }
384  }
385 
386  if( total_proc > 1 )
387  {
388  if( proc_recv_sampling != master_proc )
389  {
390  reqs[count_reqs]=Environment::worldComm().isend( proc_recv_sampling , tag, *this);
391  count_reqs++;
392  super::clear();
393  }
394  }//total_proc > 1
395 
396  }//proc_recv_sampling
397 
398  //the master proc fill its sampling in last
399  //because sampling for others are deleted via super::clear
400  if( master_proc < extra_elements )
401  {
402  total_number_of_elements_per_proc=number_of_elements_per_proc+1;
403  shift=master_proc*total_number_of_elements_per_proc;
404  }
405  else
406  {
407  int e=number_of_elements_per_proc;
408  total_number_of_elements_per_proc=e;
409  shift=extra_elements*(e+1) + (master_proc-extra_elements)*e;
410  }

412  {

415  }
416 
417  }//master proc

653  {
654  int number_of_comp = vector_components.size();
655  for(int comp=0; comp<number_of_comp; comp++)

662  for(int direction=0; direction<number_of_directions; direction++)

669  auto max=M_space->max();

679  {

682  mu(direction) = value ;
683  }
684

686  already_exist=true;

688  already_exist=true;

694  {

567 template <typename T>
568 typename type_traits<T>::real_type
569 inner_product( Vector<T> const& v1, Vector<T> const& v2 )
570 {
571  FEELPP_ASSERT( v1.localSize() == v2.localSize() &&
572  v1.size() == v2.size() )
573  ( v1.localSize() )( v2.localSize() )
574  ( v1.size() )( v2.size() ).error( "incompatible vector sizes" );
575 
576  typedef typename type_traits<T>::real_type real_type;
577 
578  size_type s = v1.localSize();
579  //size_type s = v1.map().nLocalDofWithoutGhost();
580  real_type res = 0;
581  size_type start = v1.firstLocalIndex();
582  real_type global_res = 0;
583 
584  if ( v1.comm().size() == 1 )
585  {
586  for ( size_type i = 0; i < s; ++i )
587  res += v1( start + i )* v2( start + i );
588 
589  global_res = res;
590  }
591  else
592  {
593  for ( size_type i = 0; i < s; ++i )
594  {
595  if ( !v1.localIndexIsGhost( start + i ) )
596  res += v1( start + i )* v2( start + i );
597  }
598 #if defined( FEELPP_HAS_MPI )
599  mpi::all_reduce( v1.comm(), res, global_res, std::plus<real_type>() );
600 #endif
601  }
602 
603  return global_res;
604 }

666 template <typename ElementSpaceType>
667 void
668 straightenMeshUpdateEdgesOnBoundaryIsolated( ElementSpaceType & straightener, mpl::int_<3> )
669 {
670  typedef typename ElementSpaceType::functionspace_type space_type;
671  typedef typename space_type::mesh_type mesh_type;
672  typedef typename space_type::dof_type::fe_type fe_type;
673 
674  auto const ncdof = space_type::dof_type::nComponents;
675  auto const dofshift = fe_type::nDofPerVertex*mesh_type::element_type::numVertices;
676 
677  auto mesh = straightener.mesh();
678  auto const myrank = mesh->worldComm().localRank();
679 
680  std::set<size_type> edgeIdFoundToUpdate;
681 
682  auto itedge = mesh->beginEdgeOnBoundary();
683  auto const enedge = mesh->endEdgeOnBoundary();
684  for ( ; itedge!=enedge ; ++itedge )
685  {
686  if (itedge->processId()!=myrank || itedge->numberOfElementsGhost()==0 ) continue;
687 
688  auto const theedgeid = itedge->id();
689 
690  std::set<size_type> ghostFaceIdFoundOnBoundary;
691 
692  auto iteltghost = itedge->elementsGhost().begin();
693  auto const eneltghost = itedge->elementsGhost().end();
694  for ( ; iteltghost!=eneltghost ; ++iteltghost )
695  {
696  auto const& eltGhost = mesh->element(iteltghost->template get<1>(),iteltghost->template get<0>());
697  for ( uint16_type f = 0 ; f < mesh_type::element_type::numTopologicalFaces ; ++f )
698  {
699  auto const& theface = eltGhost.face(f);
700  if ( theface.isOnBoundary() )
701  {
702  bool findEdge=false;
703  for ( uint16_type e = 0; e < mesh_type::face_type::numEdges && !findEdge ; ++e )
704  {
705  if ( theface.edge(e).id() == theedgeid) { findEdge=true; ghostFaceIdFoundOnBoundary.insert(theface.id());}
706  }
707  }
708  }
709  }
710 
711  if (ghostFaceIdFoundOnBoundary.size()==2) edgeIdFoundToUpdate.insert(theedgeid);
712 
713  } // for ( ; itedge!=enedge ; ++itedge )
714 
715 
716  if (edgeIdFoundToUpdate.size() > 0)
717  {
718  auto iteltactif = mesh->beginElementOnBoundary();
719  auto const eneltactif = mesh->endElementOnBoundary();
720  for ( ; iteltactif!=eneltactif ; ++iteltactif )
721  {
722  //if (iteltactif->processId()!=myrank) continue;
723 
724  for ( uint16_type e = 0; e < mesh_type::element_type::numEdges ; ++e )
725  {
726  if ( edgeIdFoundToUpdate.find(iteltactif->edge(e).id()) != edgeIdFoundToUpdate.end())
727  {
728  //std::cout << "find edge " << std::endl;
729  auto const idEltFind = iteltactif->id();
730  for ( uint16_type locdof = 0 ; locdof<fe_type::nDofPerEdge ; ++locdof )
731  {
732  auto const local_id = dofshift + e*fe_type::nDofPerEdge + locdof;
733 
734  for ( uint16_type comp = 0; comp < ncdof; ++comp )
735  {
736  auto const globdof = straightener.functionSpace()->dof()->localToGlobal( idEltFind, local_id, comp ).template get<0>();
737  //std::cout << straightener.functionSpace()->dof()->dofPoint( globdof ).template get<0>() << std::endl;
738  straightener(globdof) = 0;
739  }
740  }
741  }
742  }
743 
744  } // for ( ; iteltactif!=eneltactif ; ++iteltactif )
745  } // if (edgeIdFoundToUpdate.size() > 0)
746 
747 
748 } // straightenMeshUpdateEdgesOnBoundaryIsolated

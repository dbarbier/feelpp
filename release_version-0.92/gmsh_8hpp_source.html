664  _mesh_ptrtype _mesh( mesh );
665 
666  using namespace vf;
667  typedef FunctionSpace<_mesh_type,bases<Lagrange<_mesh_type::nOrder,Vectorial> > > space_t;
668 #if defined(FEELPP_ENABLE_MPI_MODE)
669  auto Xh = space_t::New( _mesh=_mesh, _worldscomm=std::vector<WorldComm>(1,worldcomm) );
670 #else
671  auto Xh = space_t::New( _mesh=_mesh );
672 #endif
673  auto xHo = vf::project( _space=Xh, _range=elements( mesh ), _expr=vf::P(), _geomap=GeomapStrategyType::GEOMAP_HO );
674  auto xLo = vf::project( _space=Xh, _range=elements( mesh ), _expr=vf::P(), _geomap=GeomapStrategyType::GEOMAP_O1 );
675  auto xHoBdy = vf::project( _space=Xh, _range=boundaryfaces( mesh ), _expr=vf::P(), _geomap=GeomapStrategyType::GEOMAP_HO );
676  auto xLoBdy = vf::project( _space=Xh, _range=boundaryfaces( mesh ), _expr=vf::P(), _geomap=GeomapStrategyType::GEOMAP_O1 );
677  auto straightener = Xh->element();
678  straightener=( xLo-xHo )-( xLoBdy-xHoBdy );
679  double norm_mean_value = integrate( _range=boundaryfaces( _mesh ), _expr=idv( straightener ) ).evaluate().norm();
680 
681  if ( norm_mean_value > 1e-12 )
682  std::cout << "the straightening process may have failed\n"
683  << "norm of component-wise mean value of displacement on the boundary should be 0"
684  << "norm_mean_value: " << norm_mean_value << "\n"
685  << "you should consider not using straightenMesh()\n"
686  << "\n";
687 
688  boost::shared_ptr<Exporter<_mesh_type,_mesh_type::nOrder> > exporter;
689 
690  if ( save )
691  {
692  exporter = Exporter<_mesh_type,_mesh_type::nOrder>::New( "gmsh"/*test_app->vm()*/, "straightener" );
693  exporter->step( 0 )->setMesh( _mesh );
694  exporter->step( 0 )->add( "xHo", xHo );
695  exporter->step( 0 )->add( "xLo", xLo );
696  exporter->step( 0 )->add( "xHoBdy", xHoBdy );
697  exporter->step( 0 )->add( "xLoBdy", xLoBdy );
698  exporter->step( 0 )->add( "straightener", straightener );
699  exporter->save();
700  }

704

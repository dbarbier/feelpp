541

547  }
548  this->M_B.block(0,0,__M,__M).template triangularView<Eigen::UnitLower>().solveInPlace(rhs);
549  LOG(INFO) << "solve B sol = rhs with rhs = " << rhs <<"\n";
550 
551  // res(:,i)=M_g(:,i)-q(:,0:i)*sigma
552  LOG(INFO) << "compute residual..." <<"\n";

556  return vf::project( _space=M_model->functionSpace(),
557  _expr=idv(M_g[__M])-idv( z ) );

602  _expr=idv(Z)-idv( expansion( M_q, rhs, __M ) ) );

605  maxerr( index++ ) = resmax.template get<0>();
606  int index2;
607  auto err = maxerr.array().abs().maxCoeff( &index2 );
608  LOG_EVERY_N(INFO, 1 ) << " (every 10 mu) maxerr=" << err << " at index = " << index2 << " at mu = " << trainset->at(index2) << "\n";
609  }
610  LOG_ASSERT( index == trainset->size() ) << "Invalid index " << index << " should be equal to trainset size = " << trainset->size() << "\n";
611  auto err = maxerr.array().abs().maxCoeff( &index );
612  LOG(INFO)<< "err=" << err << " reached at index " << index << " and mu=" << trainset->at(index) << "\n";

391 {
392  bool is_element_on_processor = false;
393 
394  // if there is no partition tag (only 2 tags) then consider all elements on
395  // current processor
396  if ( tag.size() == 2 )
397  return boost::make_tuple( true,false );
398 
399  // tag[2] is the number of partition tags (1 in general 2 if cell share an
400  // face with 2 processors)
401  int rank = this->worldComm().localRank();
402  auto it = std::find_if( tag.begin()+3, tag.end(), [&rank] ( int i )
403  {
404  return i == rank;
405  } );
406 
407  if ( it != tag.end() )
408  is_element_on_processor = true;
409 
410  bool isGhostCell=false;
411  int idOfGhostCell=0;
412 
413  if ( tag[3]!=rank )
414  {
415  isGhostCell=true;
416  idOfGhostCell=tag[3];
417  }
418 
419  return boost::make_tuple( is_element_on_processor, boost::make_tuple( isGhostCell,idOfGhostCell ) );

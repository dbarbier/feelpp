683  {

688  {

690  global_norm2 = local_norm2;

692  else
693  {

695  size_type start = this->firstLocalIndex();

909 element_product( VectorUblas<T> const& v1, VectorUblas<T> const& v2 )
910 {
911  FEELPP_ASSERT( v1.localSize() == v2.localSize() &&
912  v1.size() == v2.size() )
913  ( v1.localSize() )( v2.localSize() )
914  ( v1.size() )( v2.size() ).error( "incompatible vector sizes" );
915 
916  typedef typename type_traits<T>::real_type real_type;
917 
918  VectorUblas<real_type> _t( v1.map() );
919  size_type s = v1.localSize();
920  size_type start = v1.firstLocalIndex();
921 
922  for ( size_type i = 0; i < s; ++i )
923  _t.operator()( start+i ) = v1.operator()( start + i )* v2.operator()( start + i );
924 
925  return _t;
926 }

1014  double l2Norm( element_type const& u )

1019  double l2Norm( element_type const& u, mpl::bool_<false> )
1020  {
1021  auto mesh = model->functionSpace()->mesh();
1022  return math::sqrt( integrate( elements(mesh), (vf::idv(u))*(vf::idv(u)) ).evaluate()(0,0) );
1023  }
1024  double l2Norm( element_type const& u, mpl::bool_<true>)
1025  {
1026  auto mesh = model->functionSpace()->mesh();
1027  auto uT = u.template element<1>();
1028  return math::sqrt( integrate( elements(mesh), (vf::idv(uT))*(vf::idv(uT)) ).evaluate()(0,0) );
1029  }
1030  //double h1Norm( typename ModelType::parameter_type const& mu, int N )
1031  double h1Norm( element_type const& u )
1032  {
1033  static const bool is_composite = functionspace_type::is_composite;
1034  return h1Norm( u, mpl::bool_< is_composite >() );
1035  }
1036  double h1Norm( element_type const& u, mpl::bool_<false> )
1037  {
1038  auto mesh = model->functionSpace()->mesh();
1039  double l22 = integrate( elements(mesh), (vf::idv(u))*(vf::idv(u)) ).evaluate()(0,0);
1040  double semih12 = integrate( elements(mesh), (vf::gradv(u))*trans(vf::gradv(u)) ).evaluate()(0,0);
1041  return math::sqrt( l22+semih12 );
1042  }
1043  double h1Norm( element_type const& u, mpl::bool_<true>)

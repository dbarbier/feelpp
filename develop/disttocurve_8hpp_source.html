330  void locateElementsCrossedByCurve(bool randomlyBroadenNodesPositions=false, double randomnessAmplitude = option("gmsh.hsize").as<double>() / 2 )
331  {
332  // locate the elements crossed by the curve
333  // store their ids in a map with the "t" of the nodes being in the element
334  // update the marker2 with the elements being crossed
335 
336  CHECK( ! tNodeMap.empty() )<<"\n No nodes defining the curve have been loaded.\n";
337 
338  // create a P0 elt containing the ids of the elements
339  auto ids = M_spaceP0->element();
340  for (auto const& it : elements(M_mesh) )
341  ids.assign( it.id(), 0, 0, it.id() );
342 
343  auto ctx = M_spaceP0->context();
344 
345  std::default_random_engine re( (unsigned int)time(0) );
346  std::uniform_real_distribution<double> smallRd( -randomnessAmplitude, randomnessAmplitude );
347 
348  for (auto const& tnode : tNodeMap)
349  {
350  node_type nodeToAdd = tnode.second;
351 
352  if (randomlyBroadenNodesPositions)
353  for (int i=0; i<dim; ++i)
354  nodeToAdd[i] += smallRd(re);
355 
356  ctx.add( nodeToAdd );
357  }
358 
359  auto allIndexes = ids.evaluate( ctx );
360  const int nbPtContext = ctx.nPoints();
361  auto eltHavingPoints = M_spaceP0->element();
362 
363  auto tnodeit = tNodeMap.begin();
364  for (int i=0; i < nbPtContext; ++i )
365  {
366 
367  if (Environment::worldComm().localRank() == ctx.processorHavingPoint( i ) )
368  {
369  const double tOfNode = (*tnodeit).first;
370  const size_type index = allIndexes(i);
371 
372  if ( pointsAtIndex.count( index ) )
373  pointsAtIndex[ index ].push_back( tOfNode );
374  else
375  {
376  std::vector<double> v(1, tOfNode);
377  pointsAtIndex[ index ] = v;
378  }
379 
380  eltHavingPoints.assign( index, 0, 0, 1);
381  }
382

394  element_ptrtype makeDistanceFunctionSequential()
395  {
396  CHECK( M_dt1 >= 0 )<<"\n Intervall between the points of the curve has to be set (either the dt of the parametrized curve or an integer 1 if curve is read from a file\n";
397  if (dim==3)
398  CHECK( M_dt2 >= 0 )<<"\n Intervall between the points of the curve has to be set (either the dt of the parametrized curve or an integer 1 if curve is read from a file\n";
399 
400  const double bigdouble = 1e8;
401 
402  auto shape = M_spaceP1->elementPtr();
403  *shape = vf::project(M_spaceP1, elements(M_mesh), cst(bigdouble) );
404 
405 
406  // squared distance between a point where only its "t" is given, and a node (x,y)
407  auto distToPt = [this] (double t, double x, double y, double z) -> double
408  {
409  node_type node = this->tNodeMap[ t ];
410  if (dim==2)
411  return (x - node[0])*(x - node[0]) + (y - node[1])*(y - node[1]);
412  else
413  return (x - node[0])*(x - node[0]) + (y - node[1])*(y - node[1]) + (z - node[2])*(z - node[2]);
414  };
415 
416 
417  auto it_elt = M_mesh->elementsWithMarker2(1, M_mesh->worldComm().localRank()).first;
418  auto en_elt = M_mesh->elementsWithMarker2(1, M_mesh->worldComm().localRank()).second;
419 
420  for(; it_elt!=en_elt; it_elt++)
421  for (int j=0; j<ndofv; ++j)
422  {
423  double closestDist = bigdouble;
424  double closestPoint = bigdouble;
425 
426 
427  const size_type indexGlobDof = M_spaceP1->dof()->localToGlobal(it_elt->id(), j, 0).index();
428 
429 
430  // coords of the dof
431  const double xdof = M_spaceP1->dof()->dofPoint( indexGlobDof ).template get<0>()[0];
432  const double ydof = M_spaceP1->dof()->dofPoint( indexGlobDof ).template get<0>()[1];
433  const double zdof = dim==3 ? M_spaceP1->dof()->dofPoint( indexGlobDof ).template get<0>()[2] : 0;
434 
435  // find the point in the element having the closest distance with the dof. This distance will be the value of shape at this dof (if a smaller distance on the same dof is not found in an other element).
436  //This method assumes that the distance between the points of the curve is very small compared to the size of the mesh
437 
438  for (auto const& pt : pointsAtIndex[ it_elt->id() ] )
439  {
440  const double dtp = distToPt( pt, xdof, ydof, zdof );
441  if( dtp < closestDist )
442  {
443  closestDist = dtp;
444  closestPoint = pt;
445  }
446  }
447 
448 
449  if ( closestDist < (*shape)(indexGlobDof) * (*shape)(indexGlobDof) )
450  {
451  const node_type closestPointCoord = tNodeMap[ closestPoint ];
452  // (tx, ty) = vector tangent to the parametrized curve at the closest point on param curve
453  double t1x, t1y, t1z, t2x, t2y, t2z;
454  // try to get the point next to the closest point (in the particular case where closest point is the last point, get the previous one)
455 
456  try
457  {
458  const node_type closestPointPlusDtCoord = tNodeMap.at(closestPoint + 1.);
459  t1x = closestPointPlusDtCoord[0] - closestPointCoord[0];
460  t1y = closestPointPlusDtCoord[1] - closestPointCoord[1];
461  t1z = (dim==3) ? closestPointPlusDtCoord[2] - closestPointCoord[2] : 0;
462  }
463 
464  catch (const std::out_of_range& oor)
465  {
466  const node_type closestPointMinusDtCoord = tNodeMap.at(closestPoint - 1.);
467  t1x = closestPointCoord[0] - closestPointMinusDtCoord[0];
468  t1y = closestPointCoord[1] - closestPointMinusDtCoord[1];
469  t1z = (dim==3) ? closestPointCoord[2] - closestPointMinusDtCoord[2] : 0;
470  }
471 
472  if (dim==3)
473  {
474  try
475  {
476  const node_type closestPointPlusDtCoord = tNodeMap.at(closestPoint + periodT2);
477  t2x = closestPointPlusDtCoord[0] - closestPointCoord[0];
478  t2y = closestPointPlusDtCoord[1] - closestPointCoord[1];
479  t2z = closestPointPlusDtCoord[2] - closestPointCoord[2];
480  }
481 
482  catch (const std::out_of_range& oor)
483  {
484 
485  const node_type closestPointMinusDtCoord = tNodeMap.at(closestPoint - periodT2);
486  t2x = closestPointCoord[0] - closestPointMinusDtCoord[0];
487  t2y = closestPointCoord[1] - closestPointMinusDtCoord[1];
488  t2z = closestPointCoord[2] - closestPointMinusDtCoord[2];
489  }
490  }
491  // (vx, vy) = vector pointing from the closest point on curve to the concerned dof
492  const double vx = xdof - closestPointCoord[0];
493  const double vy = ydof - closestPointCoord[1];
494  const double vz = (dim==3) ? zdof - closestPointCoord[2] : 0;
495 
496  // the sign of the distance function is ruled by the vectorial product of the tangent vector and the vector v : sign(v x t)
497  // in 3D, it should be somthing like : sign( (v x t) . n ) where n is the normal of the param surface pointing outward
498  const double nx = t1y*t2z-t1z*t2y;
499  const double ny = t1z*t2x-t1x*t2z;
500  const double nz = t1x*t2y-t1y*t2x;
501 
502  double signProdVec;
503  if (dim==2)
504  signProdVec = vx * t1y - vy * t1x > 0 ? 1 : -1;
505  else
506  signProdVec = vx * nx + vy * ny + vz * nz > 0 ? 1 : -1;
507 
508  (*shape)( indexGlobDof ) = std::sqrt(closestDist) * signProdVec;
509 
510  }
511  }
512 
513  return shape;
514 
515  } // makeDistanceFunctionSequential

522  void reduceDistanceFunction( element_ptrtype shape )
523  {
524  // given a distance function made by makeDistanceFunctionSequential which has different values on nodes being at the interface between several subdomain, make a nice, homogeneous distance function (requires several communications though all the proc !)
525 
526  auto eltHavingPointP1 = vf::project(M_spaceP1, marked2elements(M_mesh, 1), cst(1) );
527 
528 
529  // search for all the dof being marked on at least one proc and being ghost on at least one proc (not necessarily the same proc)
530  auto checkMarked = backend()->newVector( M_spaceP1 );
531  auto checkGhost = backend()->newVector( M_spaceP1 );
532  for (size_type k = 0 ; k < M_spaceP1->nLocalDof() ; ++k )
533  {
534  checkMarked->add(k, eltHavingPointP1(k) == 1 ? 1 : 0 );
535  checkGhost->add( k, M_spaceP1->dof()->dofGlobalProcessIsGhost( k ) ? 1 : 0 );
536  }
537  checkMarked->close();
538  checkGhost->close();
539 
540 
541  // store the value of all the dof being marked and ghost
542  std::vector< std::pair< size_type, value_type > > idOnClusterAndValue;
543  for (size_type k = 0 ; k < M_spaceP1->nLocalDof() ; ++k )
544  if ( (*checkGhost)(k) && (*checkMarked)(k) )
545  idOnClusterAndValue.push_back( { processorToCluster(k), (*shape)(k) } );
546 
547 
548 
549  // gather all these values to one single proc
550  std::vector< std::vector< std::pair< size_type, value_type > > > allIdOnClusterAndValue;
551  mpi::gather( Environment::worldComm().globalComm(),
552  idOnClusterAndValue,
553  allIdOnClusterAndValue, 0 );
554 
555 
556  // this proc makes all the work :
557  // extract all the values in a map having (key = GlobalIdOnCluster, value = min value of shape )
558  std::map< size_type, value_type > idOnClusterAndMinValue;
559  if (Environment::worldComm().localRank() == 0)
560  for(auto const& v1 : allIdOnClusterAndValue)
561  for (auto const& idValue : v1)
562  {
563  if (idOnClusterAndMinValue.count( idValue.first ) )
564  idOnClusterAndMinValue[ idValue.first ] =
565  std::abs( idValue.second ) < std::abs( idOnClusterAndMinValue[ idValue.first ] )
566  ? idValue.second : idOnClusterAndMinValue[ idValue.first ];
567 
568  else
569  idOnClusterAndMinValue.insert( idValue );
570  }
571 
572  allIdOnClusterAndValue.clear(); // the info is treated, this vector is not needed anymore
573  idOnClusterAndValue.clear();
574 
575  // all proc get a copy of the id and the good min value
576  mpi::broadcast( Environment::worldComm().globalComm(),
577  idOnClusterAndMinValue, 0);
578 
579 
580 
581  // all the elements which are on the proc (ghost or not) have to be set to the min value
582  for (auto const& idValue : idOnClusterAndMinValue)
583  {
584  size_type locId=invalid_size_type_value;
585 
586  if ( (M_spaceP1->dof()->dofGlobalClusterIsOnProc( idValue.first )) )
587  locId = clusterToProcessor(idValue.first);
588 
589  else if ( ghostClusterToProc.count(idValue.first) )
590  locId = ghostClusterToProc[ idValue.first ];
591 
592  if ( locId != invalid_size_type_value )
593  (*shape)(locId) = idValue.second;
594  }
595 
596  } // reduceDistanceFunction
